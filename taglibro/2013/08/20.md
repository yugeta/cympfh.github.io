大変古い本だけど，class定義から懇切丁寧で，今まで漁った本の中でも良い感じである

```cpp
#include <iostream>
#define var auto
using namespace std;

class Hoge {
  int x;
  public:
    Hoge(int x_=0){ x = x_; }
    Hoge * print() {
      cout << x << endl;
      return this;
    }
    Hoge * inc(int d = 1) {
      x += d;
      return this;
    }
    Hoge * sq () {
      x = x*x;
      return this;
    }
};

main () {
  Hoge * a = new Hoge;
  a -> print()
    -> inc(2)
    -> print()
    -> sq()
    -> print()
  ;
}
```
