" delete opening file
fun! s:deleteThis()
    let c = input("delete this? [Y/n] > ")
    if c != "no" || c != "の"
      :!rm %
      :norm ZQ
    endif
endfun
nn <silent> <Del><Del> :call <SID>deleteThis()<cr>
