// log functions
template<class S, class T>
ostream& operator<<(ostream& os, pair<S,T> p) {
  os << '(' << car(p) << ", " << cdr(p) << ')';
  return os;
}
template<class T>
ostream& operator<<(ostream& os, vector<T> v) {
  os << "(vector";
  for (int i=0, len=v.size(); i<len; ++i) os << ' ' << v[i];
  os << ')';
  return os;
}

// array convert
template <class T>
inline
vector<T>
arr2vec(T* ar, int n) {
  vector<T> ret;
  for (int i=0; i<n; ++i) ret.push_back(ar[i]);
  return ret;
}
